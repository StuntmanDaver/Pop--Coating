    ```sql
    -- Test helpers to simulate JWT claims via set_config('request.jwt.claims', ...).
    -- Used by pgTAP tests to switch between staff_office, staff_shop, customer, anon JWTs.

    CREATE OR REPLACE FUNCTION set_jwt_for_staff(p_staff_id UUID) RETURNS void
    LANGUAGE plpgsql AS $$
    DECLARE v_staff record;
    BEGIN
      SELECT id, tenant_id, role INTO v_staff FROM staff WHERE id = p_staff_id;
      PERFORM set_config('request.jwt.claims', jsonb_build_object(
        'app_metadata', jsonb_build_object(
          'tenant_id', v_staff.tenant_id,
          'audience', CASE WHEN v_staff.role = 'shop' THEN 'staff_shop' ELSE 'staff_office' END,
          'role', v_staff.role,
          'staff_id', v_staff.id
        )
      )::text, true);
    END;
    $$;

    CREATE OR REPLACE FUNCTION set_jwt_for_customer(p_customer_user_id UUID) RETURNS void
    LANGUAGE plpgsql AS $$
    DECLARE v_cu record;
    BEGIN
      SELECT id, tenant_id, company_id INTO v_cu FROM customer_users WHERE id = p_customer_user_id;
      PERFORM set_config('request.jwt.claims', jsonb_build_object(
        'app_metadata', jsonb_build_object(
          'tenant_id', v_cu.tenant_id,
          'audience', 'customer',
          'company_id', v_cu.company_id,
          'customer_user_id', v_cu.id
        )
      )::text, true);
    END;
    $$;

    CREATE OR REPLACE FUNCTION set_jwt_for_workstation(p_workstation_id UUID) RETURNS void
    LANGUAGE plpgsql AS $$
    DECLARE v_ws record;
    BEGIN
      SELECT id, tenant_id INTO v_ws FROM workstations WHERE id = p_workstation_id;
      PERFORM set_config('request.jwt.claims', jsonb_build_object(
        'app_metadata', jsonb_build_object(
          'tenant_id', v_ws.tenant_id,
          'audience', 'staff_shop',
          'role', 'shop',
          'workstation_id', v_ws.id
        )
      )::text, true);
    END;
    $$;

    CREATE OR REPLACE FUNCTION set_jwt_anon() RETURNS void
    LANGUAGE sql AS $$
      SELECT set_config('request.jwt.claims', '{}', true);
    $$;
    ```

          'tenant-name': { type: 'string' },

          'owner-email': { type: 'string' },
